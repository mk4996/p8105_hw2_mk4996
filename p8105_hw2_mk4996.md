pols_df = read_csv(file = "./pols-month.csv",
                         na = c(".", "NA", "")) |>
  janitor::clean_names() |>
  separate(mon, into = c("year", "month", "day"), sep = "-", convert = TRUE) |>
  mutate(
    month = factor(month.name[month], levels = month.name, ordered = TRUE),
    president = case_when(
      prez_gop == 1 ~ "gop",
      prez_dem == 1 ~ "dem",
      TRUE ~ NA_character_
    )) |>
  select(-day, -prez_dem, -prez_gop)
